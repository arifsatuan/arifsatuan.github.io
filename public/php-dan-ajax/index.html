$(document).ready(function() {
  $('#submitButton').click(function() {
    var namaValue = $('#nama').val();
    var kodeValue = $('#kode').val();

    $.ajax({
      url: "<?= base_url('nama_contoller')?>",
      type: 'POST',
      data: {
        nama: namaValue,
        kode: kodeValue
      },
      success: function(response) {
        $('#result').html(response);
      },
      error: function(xhr, status, error) {
        console.log(error);
      }
    });
  });
});
